    </div><script>

    let chatCount = 0;

    const dailyLimit = 75; // Limit to 75 chats per day

    document.getElementById("chatbot-button").addEventListener("click", function() {

        document.getElementById("chatbot-container").style.display = "flex";

    });

    document.getElementById("close-chatbot").addEventListener("click", function() {

        document.getElementById("chatbot-container").style.display = "none";

    });

    document.getElementById("chatbot-send").addEventListener("click", async function() {

        if (chatCount >= dailyLimit) {

            alert("Daily chat limit reached. Please try again tomorrow.");

            return;

        }

       

        const inputElement = document.getElementById("chatbot-input");

        const messagesContainer = document.getElementById("chatbot-messages");

        const userMessage = inputElement.value.trim().toLowerCase();

        if (!userMessage) return;

       

        messagesContainer.innerHTML += `<p><strong>You:</strong> ${userMessage}</p>`;

        inputElement.value = "";

        chatCount++;

       

        if (userMessage.includes("price") || userMessage.includes("cost") || userMessage.includes("rate")) {

            messagesContainer.innerHTML += `<p><strong>Jarvis:</strong> Sorry, but I cannot provide pricing information. Please contact us directly for details.</p>`;

            return;

        }

       

        async function fetchResponse(model) {

            return fetch("https://api.openai.com/v1/chat/completions", {

                method: "POST",

                headers: {

                    "Content-Type": "application/json",

                    "Authorization": `Bearer YOUR_OPENAI_API_KEY` // Replace with your API key

                },

                body: JSON.stringify({

                    model: model,

                    messages: [

                        { role: "system", content: "You are Jarvis, a customer service chatbot for Jonny's Towing & Recovery Ltd. \n\nBusiness Information:\n- **Business Hours:** Monday - Friday, 8 AM - 6 PM. Emergency towing available 24/7.\n- **Services Offered:** Flatbed towing, roadside assistance, jump starts, lockouts, and long-distance towing.\n- **Service Areas:** Richmond, Vancouver, Burnaby, and surrounding areas.\n- **Phone Number:** (604) 398-8866.\n- **Policies:** No price estimates over chat; for quotes, call directly.\n\nYou only answer questions related to towing and roadside assistance. If a question is unrelated, politely redirect the user." },

                        { role: "user", content: userMessage }

                    ],

                    max_tokens: 75 // Shorter responses to save tokens

                })

            });

        }

       

        let response = await fetchResponse("gpt-3.5-turbo");

        let data = await response.json();

        if (!data.choices && chatCount < dailyLimit / 2) { // Only allow limited GPT-4 fallback

            response = await fetchResponse("gpt-4");

            data = await response.json();

        }

       

        const botMessage = data.choices[0].message.content;

        messagesContainer.innerHTML += `<p><strong>Jarvis:</strong> ${botMessage}</p>`;

    });

</script>
